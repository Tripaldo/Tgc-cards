    <script>
        function filterCards(rarity) {
            // Rimuovi classe active da tutti i bottoni
            document.querySelectorAll('.filter-btn').forEach(btn => btn.classList.remove('active'));
            
            // Aggiungi classe active al bottone cliccato
            event.target.classList.add('active');
            
            // Mostra/nascondi sezioni
            if (rarity === 'all') {
                document.querySelectorAll('.rarity-section').forEach(section => {
                    section.style.display = 'block';
                });
            } else {
                document.querySelectorAll('.rarity-section').forEach(section => {
                    if (section.classList.contains(rarity + '-section')) {
                        section.style.display = 'block';
                    } else {
                        section.style.display = 'none';
                    }
                });
            }
        }
        
        // Aggiungi effetti hover alle carte
        document.querySelectorAll('.card').forEach(card => {
            card.addEventListener('mouseenter', function() {
                this.style.transform = 'scale(1.05) translateY(-10px)';
                this.style.boxShadow = '0 15px 30px rgba(0,0,0,0.5)';
            });
            
            card.addEventListener('mouseleave', function() {
                this.style.transform = 'scale(1) translateY(0)';
                this.style.boxShadow = '0 8px 16px rgba(0,0,0,0.3)';
            });
        });
    </script>
